1135  void destroy(value_t t)
1136  {
1137  if (t == value_t::array || t == value_t::object)
1138  {
1139  // flatten the current json_value to a heap-allocated stack
1140  std::vector<basic_json> stack;
1141 
1142  // move the top-level items to stack
1143  if (t == value_t::array)
1144  {
1145  stack.reserve(array->size());
1146  std::move(array->begin(), array->end(), std::back_inserter(stack));
1147  }
1148  else
1149  {
1150  stack.reserve(object->size());
1151  for (auto&& it : *object)
1152  {
1153  stack.push_back(std::move(it.second));
1154  }
1155  }
1156 
1157  while (!stack.empty())
1158  {
1159  // move the last item to local variable to be processed
1160  basic_json current_item(std::move(stack.back()));
1161  stack.pop_back();
1162 
1163  // if current_item is array/object, move
1164  // its children to the stack to be processed later
1165  if (current_item.is_array())
1166  {
1167  std::move(current_item.m_value.array->begin(), current_item.m_value.array->end(), std::back_inserter(stack));
1168 
1169  current_item.m_value.array->clear();
1170  }
1171  else if (current_item.is_object())
1172  {
1173  for (auto&& it : *current_item.m_value.object)
1174  {
1175  stack.push_back(std::move(it.second));
1176  }
1177 
1178  current_item.m_value.object->clear();
1179  }
1180 
1181  // it's now safe that current_item get destructed
1182  // since it doesn't have any children
1183  }
1184  }
1185 
1186  switch (t)
1187  {
1188  case value_t::object:
1189  {
1190  AllocatorType<object_t> alloc;
1191  std::allocator_traits<decltype(alloc)>::destroy(alloc, object);

8426  {
8427  // make a working copy to apply the patch to
8428  basic_json result = *this;
8429 
8430  // the valid JSON Patch operations
8431  enum class patch_operations {add, remove, replace, move, copy, test, invalid};
8432 
8433  const auto get_op = [](const std::string & op)
8434  {
8435  if (op == "add")
8436  {
8437  return patch_operations::add;
8438  }
8439  if (op == "remove")
8440  {
8441  return patch_operations::remove;
8442  }
8443  if (op == "replace")
8444  {
8445  return patch_operations::replace;
8446  }
8447  if (op == "move")
8448  {
8449  return patch_operations::move;
8450  }
8451  if (op == "copy")
8452  {
8453  return patch_operations::copy;
8454  }
8455  if (op == "test")
8456  {
8457  return patch_operations::test;
8458  }
8459 
8460  return patch_operations::invalid;
8461  };
8462 
8463  // wrapper for "add" operation; add value at ptr
8464  const auto operation_add = [&result](json_pointer & ptr, basic_json val)
8465  {
8466  // adding to the root of the target document means replacing it
8467  if (ptr.empty())
8468  {
8469  result = val;
8470  return;
8471  }
8472 
8473  // make sure the top element of the pointer exists
8474  json_pointer top_pointer = ptr.top();
8475  if (top_pointer != ptr)
8476  {
8477  result.at(top_pointer);
8478  }
8479 
8480  // get reference to parent of JSON pointer ptr
8481  const auto last_path = ptr.back();
8482  ptr.pop_back();
8483  basic_json& parent = result[ptr];
8484 
8485  switch (parent.m_type)
8486  {
8487  case value_t::null:
8488  case value_t::object:
8489  {
8490  // use operator[] to add value
8491  parent[last_path] = val;
8492  break;
8493  }
8494 
8495  case value_t::array:
8496  {
8497  if (last_path == "-")
8498  {
8499  // special case: append to back
8500  parent.push_back(val);
8501  }
8502  else
8503  {
8504  const auto idx = json_pointer::array_index(last_path);
8505  if (JSON_HEDLEY_UNLIKELY(idx > parent.size()))
8506  {
8507  // avoid undefined behavior
8508  JSON_THROW(out_of_range::create(401, "array index " + std::to_string(idx) + " is out of range", parent));
8509  }
8510 
8511  // default case: insert add offset
8512  parent.insert(parent.begin() + static_cast<difference_type>(idx), val);
8513  }
8514  break;
8515  }
8516 
8517  // if there exists a parent it cannot be primitive
8518  case value_t::string: // LCOV_EXCL_LINE
8519  case value_t::boolean: // LCOV_EXCL_LINE
8520  case value_t::number_integer: // LCOV_EXCL_LINE
8521  case value_t::number_unsigned: // LCOV_EXCL_LINE
8522  case value_t::number_float: // LCOV_EXCL_LINE
8523  case value_t::binary: // LCOV_EXCL_LINE
8524  case value_t::discarded: // LCOV_EXCL_LINE
8525  default: // LCOV_EXCL_LINE
8526  JSON_ASSERT(false); // NOLINT(cert-dcl03-c,hicpp-static-assert,misc-static-assert) LCOV_EXCL_LINE
8527  }
8528  };
8529 
8530  // wrapper for "remove" operation; remove value at ptr
8531  const auto operation_remove = [this, &result](json_pointer & ptr)
8532  {
8533  // get reference to parent of JSON pointer ptr
8534  const auto last_path = ptr.back();
8535  ptr.pop_back();
8536  basic_json& parent = result.at(ptr);
8537 
8538  // remove child
8539  if (parent.is_object())
8540  {
8541  // perform range check
8542  auto it = parent.find(last_path);
8543  if (JSON_HEDLEY_LIKELY(it != parent.end()))
8544  {
8545  parent.erase(it);
8546  }
8547  else
8548  {
8549  JSON_THROW(out_of_range::create(403, "key '" + last_path + "' not found", *this));
8550  }
8551  }
8552  else if (parent.is_array())
8553  {
8554  // note erase performs range check
8555  parent.erase(json_pointer::array_index(last_path));
8556  }
8557  };
8558 
8559  // type check: top level value must be an array
8560  if (JSON_HEDLEY_UNLIKELY(!json_patch.is_array()))
8561  {
8562  JSON_THROW(parse_error::create(104, 0, "JSON patch must be an array of objects", json_patch));
8563  }
8564 
8565  // iterate and apply the operations
8566  for (const auto& val : json_patch)
8567  {
8568  // wrapper to get a value for an operation
8569  const auto get_value = [&val](const std::string & op,
8570  const std::string & member,
8571  bool string_type) -> basic_json &
8572  {
8573  // find value
8574  auto it = val.m_value.object->find(member);
8575 
8576  // context-sensitive error message
8577  const auto error_msg = (op == "op") ? "operation" : "operation '" + op + "'";
8578 
8579  // check if desired value is present
8580  if (JSON_HEDLEY_UNLIKELY(it == val.m_value.object->end()))
8581  {
8582  // NOLINTNEXTLINE(performance-inefficient-string-concatenation)
8583  JSON_THROW(parse_error::create(105, 0, error_msg + " must have member '" + member + "'", val));
8584  }
8585 
8586  // check if result is of type string
8587  if (JSON_HEDLEY_UNLIKELY(string_type && !it->second.is_string()))
8588  {
8589  // NOLINTNEXTLINE(performance-inefficient-string-concatenation)
8590  JSON_THROW(parse_error::create(105, 0, error_msg + " must have string member '" + member + "'", val));
8591  }
8592 
8593  // no error: return value
8594  return it->second;
8595  };
8596 
8597  // type check: every element of the array must be an object
8598  if (JSON_HEDLEY_UNLIKELY(!val.is_object()))
8599  {
8600  JSON_THROW(parse_error::create(104, 0, "JSON patch must be an array of objects", val));
8601  }
8602 
8603  // collect mandatory members
8604  const auto op = get_value("op", "op", true).template get<std::string>();
8605  const auto path = get_value(op, "path", true).template get<std::string>();
8606  json_pointer ptr(path);
8607 
8608  switch (get_op(op))
8609  {
8610  case patch_operations::add:
8611  {
8612  operation_add(ptr, get_value("add", "value", false));
8613  break;
8614  }
8615 
8616  case patch_operations::remove:
8617  {
8618  operation_remove(ptr);
8619  break;
8620  }
8621 
8622  case patch_operations::replace:
8623  {
8624  // the "path" location must exist - use at()
8625  result.at(ptr) = get_value("replace", "value", false);
8626  break;
8627  }
8628 
8629  case patch_operations::move:
8630  {
8631  const auto from_path = get_value("move", "from", true).template get<std::string>();
8632  json_pointer from_ptr(from_path);
8633 
8634  // the "from" location must exist - use at()
8635  basic_json v = result.at(from_ptr);
8636 
8637  // The move operation is functionally identical to a
8638  // "remove" operation on the "from" location, followed
8639  // immediately by an "add" operation at the target
8640  // location with the value that was just removed.
8641  operation_remove(from_ptr);
8642  operation_add(ptr, v);
8643  break;
8644  }
8645 
8646  case patch_operations::copy:
8647  {
8648  const auto from_path = get_value("copy", "from", true).template get<std::string>();
8649  const json_pointer from_ptr(from_path);
8650 
8651  // the "from" location must exist - use at()
8652  basic_json v = result.at(from_ptr);
8653 
8654  // The copy is functionally identical to an "add"
8655  // operation at the target location using the value
8656  // specified in the "from" member.
8657  operation_add(ptr, v);
8658  break;
8659  }
8660 
8661  case patch_operations::test:
8662  {
8663  bool success = false;
8664  JSON_TRY
8665  {
8666  // check if "value" matches the one at "path"
8667  // the "path" location must exist - use at()
8668  success = (result.at(ptr) == get_value("test", "value", false));
8669  }
8670  JSON_INTERNAL_CATCH (out_of_range&)
8671  {
8672  // ignore out of range errors: success remains false
8673  }
8674 
8675  // throw an exception if test fails
8676  if (JSON_HEDLEY_UNLIKELY(!success))
8677  {
8678  JSON_THROW(other_error::create(501, "unsuccessful: " + val.dump(), val));
8679  }
8680 
8681  break;
8682  }
8683 
8684  case patch_operations::invalid:
8685  default:
8686  {
8687  // op must be "add", "remove", "replace", "move", "copy", or
8688  // "test"
8689  JSON_THROW(parse_error::create(105, 0, "operation value '" + op + "' is invalid", val));

8730  JSON_HEDLEY_WARN_UNUSED_RESULT
8731  static basic_json diff(const basic_json& source, const basic_json& target,
8732  const std::string& path = "")
8733  {
8734  // the patch
8735  basic_json result(value_t::array);
8736 
8737  // if the values are the same, return empty patch
8738  if (source == target)
8739  {
8740  return result;
8741  }
8742 
8743  if (source.type() != target.type())
8744  {
8745  // different types: replace value
8746  result.push_back(
8747  {
8748  {"op", "replace"}, {"path", path}, {"value", target}
8749  });
8750  return result;
8751  }
8752 
8753  switch (source.type())
8754  {
8755  case value_t::array:
8756  {
8757  // first pass: traverse common elements
8758  std::size_t i = 0;
8759  while (i < source.size() && i < target.size())
8760  {
8761  // recursive call to compare array values at index i
8762  auto temp_diff = diff(source[i], target[i], path + "/" + std::to_string(i));
8763  result.insert(result.end(), temp_diff.begin(), temp_diff.end());
8764  ++i;
8765  }
8766 
8767  // i now reached the end of at least one array
8768  // in a second pass, traverse the remaining elements
8769 
8770  // remove my remaining elements
8771  const auto end_index = static_cast<difference_type>(result.size());
8772  while (i < source.size())
8773  {
8774  // add operations in reverse order to avoid invalid
8775  // indices
8776  result.insert(result.begin() + end_index, object(
8777  {
8778  {"op", "remove"},
8779  {"path", path + "/" + std::to_string(i)}
8780  }));
8781  ++i;
8782  }
8783 
8784  // add other remaining elements
8785  while (i < target.size())
8786  {
8787  result.push_back(
8788  {
8789  {"op", "add"},
8790  {"path", path + "/-"},
8791  {"value", target[i]}
8792  });
8793  ++i;
8794  }
8795 
8796  break;
8797  }
8798 
8799  case value_t::object:
8800  {
8801  // first pass: traverse this object's elements
8802  for (auto it = source.cbegin(); it != source.cend(); ++it)
8803  {
8804  // escape the key name to be used in a JSON patch
8805  const auto path_key = path + "/" + detail::escape(it.key());
8806 
8807  if (target.find(it.key()) != target.end())
8808  {
8809  // recursive call to compare object values at key it
8810  auto temp_diff = diff(it.value(), target[it.key()], path_key);
8811  result.insert(result.end(), temp_diff.begin(), temp_diff.end());
8812  }
8813  else
8814  {
8815  // found a key that is not in o -> remove it
8816  result.push_back(object(
8817  {
8818  {"op", "remove"}, {"path", path_key}
8819  }));
8820  }
8821  }
8822 
8823  // second pass: traverse other object's elements
8824  for (auto it = target.cbegin(); it != target.cend(); ++it)
8825  {
8826  if (source.find(it.key()) == source.end())
8827  {
8828  // found a key that is not in this -> add it
8829  const auto path_key = path + "/" + detail::escape(it.key());
8830  result.push_back(
8831  {
8832  {"op", "add"}, {"path", path_key},
8833  {"value", it.value()}
8834  });
8835  }
8836  }
8837 
8838  break;
8839  }
8840 
8841  case value_t::null:
8842  case value_t::string:
8843  case value_t::boolean:
8844  case value_t::number_integer:
8845  case value_t::number_unsigned:
8846  case value_t::number_float:
8847  case value_t::binary:
8848  case value_t::discarded:
8849  default:
8850  {
8851  // both primitive type: replace value
8852  result.push_back(
8853  {
8854  {"op", "replace"}, {"path", path}, {"value", target}
8855  });
8856  break;
8857  }
8858  }
8859 
8860  return result;
8861  }
